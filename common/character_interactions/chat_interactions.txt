send_letter_interaction = {
	category = interaction_category_diplomacy
	use_diplomatic_range = no
	interface_priority = 99

	auto_accept = yes

	is_shown = {
		scope:actor = {
			is_ai = no
		}
		scope:recipient = {
			is_ai = no
		}
		NOT = {
			scope:actor = scope:recipient
		}
	}

	on_send = {
		scope:recipient = {
			save_scope_as = letter_recipient
		}

		scope:actor = {
			trigger_event = {
				id = letter.1
			}
		}
	}
}
