is_lopc_adventurer_courtier = {
	custom_description = lopc_adventurer_courtier_trigger
	trigger_if = {
		OR = {
			has_character_modifier = lopc_courtier_0
			has_character_modifier = lopc_courtier_1
			has_character_modifier = lopc_courtier_2
			has_character_modifier = lopc_courtier_3
			has_character_modifier = lopc_courtier_4
		}
	}
	trigger_else = { always = no }
}
is_lopc_adventurer_special_courtier = {
	custom_description = lopc_adventurer_courtier_special_trigger
	trigger_if = {
		OR = {
			has_character_modifier = lopc_adventurer
			has_character_modifier = lopc_holy_order
			has_character_modifier = lopc_merc
		}
	}
	trigger_else = { always = no }
}
is_lopc_low_employed = {
	custom_description = lopc_low_employment_trigger
	trigger_if = {
		OR = {
			has_character_modifier = lopc_job_guard
			has_character_modifier = lopc_job_scribe
			has_character_modifier = lopc_job_assistant
			has_character_modifier = lopc_job_ruffian
			has_character_modifier = lopc_job_herald
		}
	}
	trigger_else = { always = no }
}
lopc_is_employed = {
	custom_description = lopc_any_employment_trigger
	trigger_if = {
		OR = {
			is_lopc_low_employed = yes
			is_lopc_adventurer_special_courtier = yes
			is_lopc_adventurer_courtier = yes
		}
	}
	trigger_else = { always = no }

}
lopc_homeowner = {
	custom_description = lopc_adventurer_courtier_special_trigger
	trigger_if = {
		OR = {
			has_character_modifier = lopc_owns_home_1
			has_character_modifier = lopc_owns_home_2
			has_character_modifier = lopc_owns_home_3
			has_character_modifier = lopc_owns_home_4
		}
	}
	trigger_else = { always = no }
}
is_lopc_with_employer = {
	custom_description = = {
		text = "Local Ruler Is Character's Employer"
		has_relation_lopc_employer = root.location.barony.holder
		root.location = root.location.barony.holder.capital_province
	}
}
has_lopc_court_position = {
	trigger_if = {
		has_any_court_position = yes
	}
	trigger_else = { always = no }
}
lopc_needs_treatment = {
	custom_tooltip = {
		text = lopc_needs_treatment.tt
		OR = {
			AND = {
				has_recent_treatment_trigger = no
				has_treatable_disease_trigger = yes
			}
			AND = {
				has_recent_wound_treatment_trigger = no
				has_wounds_trigger = yes
			}
		}
	}
}
lopc_request_court_position_requirements = {
	trigger_if = {
		limit = { has_any_court_position = yes }
		custom_tooltip = {
			text = lopc_promotion_cost_loc_02 
			prestige >= 200
		}
	}
			
	trigger_else = {
		custom_tooltip = {
			text = lopc_promotion_cost_loc_01 
			prestige >= 100
		}
	}
}
