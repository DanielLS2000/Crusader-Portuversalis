# CoA trigger tests

coa_switch_to_mena_trigger = {
	holder = {
		NOT = {
			faith.religion = religion:christianity_religion
		}
		OR = {
			faith.religion = religion:islam_religion
			OR = {			
				culture = { has_coa_gfx = arabic_group_coa_gfx }
				culture = { has_coa_gfx = berber_group_coa_gfx }
			}
		}
	}
}

coa_switch_to_islamic_trigger = {
	holder = {
		faith.religion = religion:islam_religion
	}
}

coa_switch_reconquista_trigger = {
	holder = {
		NOT = {
			faith.religion = religion:islam_religion
		}	
		faith.religion = religion:christianity_religion
	}
}

ps_coa_venetian_trigger = { # Maybe limit to Venice rather than cisalpine
	OR = {
		scope:culture = { has_coa_gfx = latin_group_coa_gfx }
		AND = {
			exists = scope:title
			scope:title = {
				any_this_title_or_de_jure_above = {
					this = title:k_venice
				}
			}
		}
	}
}

		government_has_flag = government_is_theocracy
	}
}

ps_coa_switch_to_pagan_trigger = {
	holder = {
		NOT = {
			OR = {
				faith.religion = religion:christianity_religion
				faith.religion = religion:germanic_religion
				culture = { has_coa_gfx = norse_coa_gfx }
			}
		}	
		OR = {
			faith.religion = religion:paganism_religion
			faith.religion = religion:finno_ugric_religion
			faith.religion = religion:baltic_religion
			faith.religion = religion:slavic_religion
		}
	}
}

ps_coa_switch_to_norse_trigger = {
	holder = {
		NOT = {
			OR = {		
				faith.religion = religion:christianity_religion
				faith.religion = religion:paganism_religion
				faith.religion = religion:finno_ugric_religion
				faith.religion = religion:baltic_religion
				faith.religion = religion:slavic_religion
			}
		}
		OR = {
			faith.religion = religion:germanic_religion
			culture = { has_coa_gfx = norse_coa_gfx }
		}
	}
}
