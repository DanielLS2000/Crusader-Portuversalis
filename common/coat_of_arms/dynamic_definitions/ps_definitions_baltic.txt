	item = {
		trigger = {
			holder = {
				NOT = {

				}		
				OR = {
					faith.religion = religion:paganism_religion
					faith.religion = religion:finno_ugric_religion
					faith.religion = religion:baltic_religion
					faith.religion = religion:slavic_religion
				}
			}
		}
		coat_of_arms = e_wendish_empire_pagan
	}		
}
