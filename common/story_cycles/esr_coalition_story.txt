	effect_group = {
		months = 1

		triggered_effect = {

			}

			effect = {
				if = {
					limit = {
						has_variable_list = esr_threatened_chars
					}

						variable = esr_threatened_chars
						root = {
							add_to_variable_list = {
								name = esr_coalition_member
								target = prev
							}
						}
					}
				}
				set_variable = {
					name = sc_relative_power_of_coalition
					value = sc_relative_power_of_coalition
				}
			}
		}
	}
