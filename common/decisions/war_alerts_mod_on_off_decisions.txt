				name = notify_of_war_family

				name = notify_of_war_dynasty
				value = 2
			}
			set_global_variable = {
				name = notify_of_war_vassal_wars
				value = 2
			}
			set_global_variable = {
				name = notify_of_neighbour_wars
				value = 1
			}			
		}
	}
	
	ai_will_do = {
		base = 0
	}

				value = 0

	}
