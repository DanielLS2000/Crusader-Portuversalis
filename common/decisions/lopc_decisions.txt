lopc_attend_court_decision = { # goes to main court event
	desc = lopc_attend_court_decision_desc
	selection_tooltip = lopc_attend_court_decision_tooltip
	confirm_text = lopc_attend_court_decision_confirm
	picture = {
		reference = "gfx/interface/illustrations/decisions/ep3_cerimonial_decision.dds"
	}
	
	major = no
	decision_group_type = adventurer_minor
	sort_order = 50
	
	is_shown = {
		is_ai = no
		is_lopc_adventurer_courtier = yes
	}
	
	is_valid = {
		#always = yes
		OR = {
			is_travelling = no
			custom_tooltip = {
				text = join_court_interaction_correctlocation_tt
				is_lopc_with_employer = yes
			}
		}
		
	}
	
	cooldown = { months = 2 }
	
	ai_potential = {
		always = no
	}
	
	effect = {
		random_relation = {
			type = lopc_employer
			save_scope_as = courtier_x_liege
		}
		
		if = {
			limit = { NOT = { location = scope:courtier_x_liege.capital_province } } 
				
			start_travel_plan = {
				destination = scope:courtier_x_liege.capital_province
				on_arrival_event = lopc_noble_family_event.0001
			}
		}
		else = {
			trigger_event = { id = lopc_event.2100 }
		}
		#trigger_event = { id = lopc_event.2100 }
	}
}
