            reb_distance_opinion_value < 0

            limit = {
                has_game_rule = reb_opinion_33
            }
            multiply = 0.33
        }
        else_if = {
            limit = {
                has_game_rule = reb_opinion_66
            }

        else_if = {

                has_game_rule = reb_opinion_150

            multiply = 0.66

        else_if = {
