lopc_performance_value = {
    value = 0

    if = {
        limit = {
            has_variable = lopc_court_performance
        }
        #add = root.var:lopc_court_performance
		add = 1
		multiply = var:lopc_court_performance
    }

    min = 1
    desc = MODIFIER_DEFINITION_LOPC_COURT_PERFORMANCE
}
lopc_business_value = {
    value = stewardship

    if = {
        limit = {
            has_variable = lopc_home_location
        }
		add = var:lopc_home_location.county.development_level
		if = {
			limit = { has_character_modifier = lopc_business_trade_post_mixed } #should make less gold based on development, but much more otherwise
			max = 10
		}
    }
    if = {
        limit = {
            has_character_modifier = lopc_owns_home_2
        }
		multiply = 1.25
    }
    if = {
        limit = {
            has_character_modifier = lopc_owns_home_3
        }
		multiply = 1.5
    }
    if = {
        limit = {
            has_character_modifier = lopc_owns_home_4
        }
		multiply = 1.75
    }
    if = {
        limit = {
            has_character_modifier = lopc_managed_business
        }
		multiply = {
			value = stewardship
			divide = 100
			add = 1
		}
    }

    min = 1
    desc = MODIFIER_DEFINITION_LOPC_BUSINESS_VALUE
}
lopc_republican_election_value = {
    value = 0
	add = {
		value = stewardship
		multiply = 3
	}
	add = diplomacy
	add = {
		value = 20
		multiply = prestige_level
	}
	if = {
		limit = {
			has_faith = scope:courtier_x_liege.capital_province.faith
		}
		add = {
			value = 10
			multiply = piety_level
		}
	}
    if = {
        limit = {
            has_character_flag = flag:lopc_big_election_investment
        }
		add = 75
    }
    if = {
        limit = {
            has_character_flag = flag:lopc_medium_election_investment
        }
		add = 50
    }
    if = {
        limit = {
            has_character_flag = flag:lopc_small_election_investment
        }
		add = 25
    }
    if = {
        limit = {
            gold >= 500
        }
		add = 50
    }
    if = {
        limit = {
            gold >= 100
        }
		add = 50
    }
    if = {
        limit = {
            gold < 0
        }
		subtract = 200
    }
    min = 0
}
