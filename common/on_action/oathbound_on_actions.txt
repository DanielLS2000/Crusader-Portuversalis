quarterly_playable_pulse = {
	on_actions = {
		add_trait_check
		remove_trait_check
		remove_permanent_oath_check
	}
}

add_trait_check = {
	trigger = {
		AND = {
			NOT = { has_trait = oathkeeper }
			num_of_relation_oathholder > 0
		}
	}
	effect = {
		add_trait = oathkeeper
	}
}

remove_trait_check = {
	trigger = {
		AND = {
			has_trait = oathkeeper
			num_of_relation_oathholder = 0
		}
	}
	effect = {
		remove_trait = oathkeeper
	}
}

remove_permanent_oath_check = {
	trigger = {
		AND = {
			has_character_modifier = permanent_oath
			num_of_relation_oathholder = 0
		}
	}
	effect = {
		remove_character_modifier = permanent_oath
	}
}
